# FiveM ESX Admin Menu — Full Resource Skeleton (Tailwind + jQuery + oxmysql)

> Drop-in resource providing Player, Vehicle, Main, Settings, Personal menus with database-backed bans/warns/admins/languages. Built for ESX, uses NUI with TailwindCSS and jQuery.

## sql/admin_menu.sql
```sql
-- Admins table (permissions)
CREATE TABLE IF NOT EXISTS admin_menu_admins (
  id INT AUTO_INCREMENT PRIMARY KEY,
  identifier VARCHAR(60) NOT NULL,
  role ENUM('owner','admin','mod') NOT NULL DEFAULT 'mod',
  added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY (identifier)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Bans
CREATE TABLE IF NOT EXISTS admin_menu_bans (
  id INT AUTO_INCREMENT PRIMARY KEY,
  identifier VARCHAR(60) NOT NULL,
  name VARCHAR(64) NULL,
  reason VARCHAR(255) NOT NULL,
  banner_identifier VARCHAR(60) NOT NULL,
  banner_name VARCHAR(64) NULL,
  expires_at DATETIME NULL, -- NULL = permanent
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX (identifier),
  INDEX (expires_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Warns
CREATE TABLE IF NOT EXISTS admin_menu_warns (
  id INT AUTO_INCREMENT PRIMARY KEY,
  identifier VARCHAR(60) NOT NULL,
  name VARCHAR(64) NULL,
  reason VARCHAR(255) NOT NULL,
  warner_identifier VARCHAR(60) NOT NULL,
  warner_name VARCHAR(64) NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX (identifier)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Languages
CREATE TABLE IF NOT EXISTS admin_menu_languages (
  code VARCHAR(8) PRIMARY KEY,
  label VARCHAR(32) NOT NULL,
  data JSON NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Hotkeys (per-admin)
CREATE TABLE IF NOT EXISTS admin_menu_hotkeys (
  id INT AUTO_INCREMENT PRIMARY KEY,
  identifier VARCHAR(60) NOT NULL,
  key_open VARCHAR(12) NOT NULL DEFAULT 'F2',
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE KEY (identifier)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

## server/utils.lua

```lua
local M = {}

function M.GetIdentifier(src)
  local xPlayer = ESX.GetPlayerFromId(src)
  if not xPlayer then return nil end
  return xPlayer.identifier
end

function M.IsExpired(dt)
  if not dt then return false end
  return (os.time() > os.time({
    year=tonumber(dt:sub(1,4)), month=tonumber(dt:sub(6,7)), day=tonumber(dt:sub(9,10)),
    hour=tonumber(dt:sub(12,13)) or 0, min=tonumber(dt:sub(15,16)) or 0, sec=tonumber(dt:sub(18,19)) or 0
  }))
end

return M
```

## server/permissions.lua
```lua
local Perm = {}

function Perm.GetRole(identifier, cb)
  MySQL.single('SELECT role FROM admin_menu_admins WHERE identifier = ?', { identifier }, function(row)
    cb(row and row.role or nil)
  end)
end

function Perm.RequireRole(src, needed, cb)
  local id = exports['admin_menu']:GetIdentifier(src) or exports['admin_menu'].GetIdentifier(src) -- fallback pattern
  if not id then return cb(false) end
  Perm.GetRole(id, function(role)
    if not role then return cb(false) end
    local order = { mod=1, admin=2, owner=3 }
    cb(order[role] >= order[needed])
  end)
end

exports('GetRole', function(identifier, cb) Perm.GetRole(identifier, cb) end)
return Perm
```

## server/ban.lua
```lua
local Utils = require 'server/utils'
local Ban = {}

-- Add Ban
RegisterNetEvent('admin_menu:banPlayer', function(targetSrc, minutes, reason)
  local src = source
  local bannerId = Utils.GetIdentifier(src)
  if not bannerId then return end

  exports['admin_menu-perm']:Require(src, 'admin', function(ok) -- optional external wrapper
    if ok == false then return end
  end)

  local target = ESX.GetPlayerFromId(targetSrc)
  if not target then return end

  local expires = nil
  if minutes and minutes > 0 then
    expires = os.date('%Y-%m-%d %H:%M:%S', os.time() + (minutes * 60))
  end

  MySQL.insert('INSERT INTO admin_menu_bans (identifier,name,reason,banner_identifier,banner_name,expires_at) VALUES (?,?,?,?,?,?)',
    { target.identifier, target.getName() or target.get("firstName") or 'Unknown', reason or 'N/A', bannerId, GetPlayerName(src), expires },
    function()
      DropPlayer(targetSrc, ('Banned: %s | Until: %s'):format(reason or 'N/A', expires or 'Permanent'))
      TriggerClientEvent('admin_menu:notify', -1, ('%s has been banned.'):format(GetPlayerName(targetSrc)))
    end)
end)

-- Check on connect
AddEventHandler('playerConnecting', function(name, setKick, def)
  local src = source
  local ids = GetPlayerIdentifiers(src)[1]
  if not ids then return end
  local row = MySQL.single.await('SELECT * FROM admin_menu_bans WHERE identifier = ? ORDER BY id DESC LIMIT 1', { ids })
  if row then
    if row.expires_at and Utils.IsExpired(row.expires_at) then
      return
    end
    setKick(('You are banned. Reason: %s | Until: %s'):format(row.reason, row.expires_at or 'Permanent'))
  end
end)

-- Banlist search
ESX.RegisterServerCallback('admin_menu:getBanlist', function(src, cb, query)
  query = ('%%%s%%'):format(query or '')
  local rows = MySQL.query.await([[SELECT * FROM admin_menu_bans WHERE identifier LIKE ? OR name LIKE ? ORDER BY created_at DESC LIMIT 200]], { query, query })
  cb(rows or {})
end)

return Ban
```

```lua
-- Player listing + actions (give item/money, job, heal/revive/teleport/bring/freeze/dm/open inventory)

-- cache items/jobs
local cached = { items = { at = 0, list = {} }, jobs = { at = 0, list = {} } }

ESX.RegisterServerCallback('admin_menu:getOnlinePlayers', function(src, cb)
  local players = ESX.GetExtendedPlayers()
  local list = {}
  for _,xPlayer in pairs(players) do
    list[#list+1] = { id = xPlayer.source, name = GetPlayerName(xPlayer.source) or ('[%s]'):format(xPlayer.source), job = xPlayer.job and xPlayer.job.name, grade = xPlayer.job and xPlayer.job.grade }
  end
  cb(list)
end)

local function refreshItems()
  if (cached.items.at + Config.ItemRefreshMS) > GetGameTimer() then return cached.items.list end
  cached.items.at = GetGameTimer()
  cached.items.list = {}
  for name,item in pairs(exports['ox_inventory']:Items() or {}) do
    table.insert(cached.items.list, { name = name, label = item.label or name })
  end
  return cached.items.list
end

ESX.RegisterServerCallback('admin_menu:getItems', function(src, cb)
  cb(refreshItems())
end)

-- Jobs
local function refreshJobs()
  if (cached.jobs.at + Config.JobRefreshMS) > GetGameTimer() then return cached.jobs.list end
  cached.jobs.at = GetGameTimer()
  cached.jobs.list = {}
  for jobName,job in pairs(ESX.Jobs or {}) do
    local grades = {}
    for grade, gData in pairs(job.grades or {}) do
      table.insert(grades, { grade = tonumber(grade), label = gData.label or tostring(grade) })
    end
    table.sort(grades, function(a,b) return a.grade < b.grade end)
    table.insert(cached.jobs.list, { name = jobName, label = job.label or jobName, grades = grades })
  end
  return cached.jobs.list
end

ESX.RegisterServerCallback('admin_menu:getJobs', function(src, cb)
  cb(refreshJobs())
end)

-- Give Items (multiple)
RegisterNetEvent('admin_menu:giveItems', function(targetSrc, items)
  local xTarget = ESX.GetPlayerFromId(targetSrc)
  if not xTarget or type(items) ~= 'table' then return end
  for _, it in ipairs(items) do
    if it.name and it.count and it.count > 0 then
      xTarget.addInventoryItem(it.name, it.count)
    end
  end
  TriggerClientEvent('admin_menu:notify', targetSrc, 'Items received')
end)

-- Give Money
RegisterNetEvent('admin_menu:giveMoney', function(targetSrc, account, amount)
  local xTarget = ESX.GetPlayerFromId(targetSrc)
  if not xTarget or not account or not amount then return end
  xTarget.addAccountMoney(account, amount)
  TriggerClientEvent('admin_menu:notify', targetSrc, ('You received %s: %s'):format(account, amount))
end)

-- Set job & grade
RegisterNetEvent('admin_menu:setJob', function(targetSrc, job, grade)
  local xTarget = ESX.GetPlayerFromId(targetSrc)
  if not xTarget then return end
  xTarget.setJob(job, grade or 0)
end)

RegisterNetEvent('admin_menu:setJobGrade', function(targetSrc, grade)
  local xTarget = ESX.GetPlayerFromId(targetSrc)
  if not xTarget or not xTarget.job then return end
  xTarget.setJob(xTarget.job.name, grade)

-- Heal/Revive
RegisterNetEvent('admin_menu:heal', function(targetSrc)
  TriggerClientEvent('admin_menu:c:heal', targetSrc)
end)
RegisterNetEvent('admin_menu:revive', function(targetSrc)
  TriggerClientEvent('admin_menu:c:revive', targetSrc)
end)

-- Teleport/Bring

RegisterNetEvent('admin_menu:goto', function(src, targetSrc)
```

```lua
local noclip = false
local invisible = false
local superJump = false
local fastRun = false
local ghost = false

local function ToggleNoClip()
  noclip = not noclip
  local ped = PlayerPedId()
  SetEntityInvincible(ped, noclip)
  SetEntityCollision(ped, not noclip, not noclip)
  SetEntityVisible(ped, not noclip, false)
  CreateThread(function()
    while noclip do
      local p = GetEntityCoords(ped)
      local fwd = GetEntityForwardVector(ped)
      local spd = IsControlPressed(0, 21) and 2.0 or 0.8
      if IsControlPressed(0, 32) then p = p + fwd * spd end -- W
      if IsControlPressed(0, 33) then p = p - fwd * spd end -- S
      if IsControlPressed(0, 34) then p = p + vector3(-fwd.y, fwd.x, 0.0) * spd end -- A
      if IsControlPressed(0, 35) then p = p + vector3(fwd.y, -fwd.x, 0.0) * spd end -- D
      if IsControlPressed(0, 22) then p = p + vector3(0,0,spd) end -- Space
      if IsControlPressed(0, 36) then p = p - vector3(0,0,spd) end -- Ctrl
      SetEntityCoordsNoOffset(ped, p.x, p.y, p.z, true, true, true)
      Wait(0)
    end
  end)
end

local function ToggleInvisible()
  invisible = not invisible
  SetEntityVisible(PlayerPedId(), not invisible, false)
end

CreateThread(function()
  while true do
    Wait(0)
    if superJump then SetSuperJumpThisFrame(PlayerId()) end
    if fastRun then SetRunSprintMultiplierForPlayer(PlayerId(), 1.49) else SetRunSprintMultiplierForPlayer(PlayerId(), 1.0) end
    if ghost then SetEntityCollision(PlayerPedId(), false, false) else SetEntityCollision(PlayerPedId(), true, true) end
  end
end)

RegisterNetEvent('admin_menu:toggle:noclip', ToggleNoClip)
RegisterNetEvent('admin_menu:toggle:invisible', ToggleInvisible)
RegisterNetEvent('admin_menu:toggle:superjump', function(state) superJump = state end)
RegisterNetEvent('admin_menu:toggle:fastrun', function(state) fastRun = state end)
RegisterNetEvent('admin_menu:toggle:ghost', function(state) ghost = state end)
```

```js
const Nui = (name, data) => new Promise((res) => $.post(`https://${GetParentResourceName()}/${name}`, JSON.stringify(data||{}), res));

const ui = {
  open(){ $('#root').removeClass('hidden-ui'); },
  close(){ $('#root').addClass('hidden-ui'); Nui('close'); },
  notify(msg){ /* toast: simple */ alert(msg) }
};

window.addEventListener('message', (e) => {
  const { action } = e.data || {};
  if(action === 'open') ui.open();
  if(action === 'notify') ui.notify(e.data.message);
});

$(function(){
  // Tabs
  $('#tabs button').on('click', function(){
    const tab = $(this).data('tab');
    $('[id^=view-]').addClass('hidden-ui');
    $(`#view-${tab}`).removeClass('hidden-ui');
  });

  $('#btn-close').on('click', ui.close);
  $(document).on('keydown', (e)=>{ if(e.key==='Escape') ui.close(); });

  // Load players
  const loadPlayers = async () => {
    const list = await Nui('players:list');
    const $wrap = $('#player-list').empty();
    list.forEach(p => {
      const card = $(`<div class='p-2 bg-slate-700 rounded-xl flex items-center justify-between'>
        <div>
          <div class='font-semibold'>${p.name}</div>
          <div class='text-xs opacity-75'>ID: ${p.id} • ${p.job||'-'} ${p.grade??''}</div>
        </div>
        <input type='radio' name='target' value='${p.id}' />
      </div>`);
      $wrap.append(card);
    });
  };
  $('#refresh-players').on('click', loadPlayers);
  loadPlayers();

  // Items
  const loadItems = async () => {
    const items = await Nui('players:getitems');
    const $list = $('#item-list').empty();
    items.forEach(it => {
      const row = $(`<div class='flex items-center gap-2 py-1'>
        <input type='checkbox' value='${it.name}' />
        <span class='flex-1'>${it.label}</span>
        <input type='number' min='1' class='input w-20' placeholder='x'/>
      </div>`);
      $list.append(row);
    });
  };
  Nui('players:getitems').then(loadItems);

  // Jobs
  const loadJobs = async () => {
    const jobs = await Nui('players:getjobs');
    const $job = $('#job-select').empty();
    jobs.forEach(j => { $job.append(`<option value='${j.name}'>${j.label}</option>`); });
    $job.on('change', function(){
      const j = jobs.find(x=>x.name===this.value);
      const $g = $('#job-grade').empty();
      (j.grades||[]).forEach(g => $g.append(`<option value='${g.grade}'>${g.label}</option>`));
    }).trigger('change');
  };
  loadJobs();

  // Accounts
  const accounts = [ {name:'money',label:'Cash'}, {name:'bank',label:'Bank'} ];
  accounts.forEach(a => $('#money-account').append(`<option value='${a.name}'>${a.label}</option>`));

  const getTarget = () => parseInt($('input[name=target]:checked').val());

  // Give Items
  $('#btn-give-items').on('click', async () => {
    const target = getTarget();
    const items = [];
    $('#item-list > div').each(function(){
      const name = $(this).find('input[type=checkbox]').val();
      const checked = $(this).find('input[type=checkbox]').is(':checked');
      const count = parseInt($(this).find('input[type=number]').val());
      if(checked && count>0) items.push({name, count});
    });
    await Nui('players:giveitems', { target, items });
  });

  // Give Money
  $('#btn-give-money').on('click', async () => {
    await Nui('players:givemoney', { target: getTarget(), account: $('#money-account').val(), amount: parseInt($('#money-amount').val()) });
  });

  $('#btn-set-job').on('click', async () => {
    await Nui('players:setjob', { target: getTarget(), job: $('#job-select').val(), grade: parseInt($('#job-grade').val()) });
  });
  $('#btn-set-grade').on('click', async () => {
    await Nui('players:setjobgrade', { target: getTarget(), grade: parseInt($('#job-grade').val()) });
  });

  // Quick actions
  $('[data-act]').on('click', async function(){
    const act = $(this).data('act');
    const target = getTarget();
    if(!target && act!=='dm') return alert('Select target');
    if(act==='heal') await Nui('players:heal', { target });
    if(act==='revive') await Nui('players:revive', { target });
    if(act==='kill') await Nui('players:kill', { target });
    if(act==='goto') await Nui('players:goto', { src: -1, target });
    if(act==='bring') await Nui('players:bring', { target, to: -1 });
    if(act==='spectate') await Nui('players:spectate', { target });
    if(act==='freeze') await Nui('players:freeze', { target, state: true });
    if(act==='unfreeze') await Nui('players:freeze', { target, state: false });
    if(act==='openinv') await Nui('players:openinv', { target });
    if(act==='dm') {
      const msg = prompt('Message');
      await Nui('players:dm', { target, msg });
    }
    if(act==='warn') {
      const reason = prompt('Reason');
      await Nui('players:warn', { target, reason });
    }
    if(act==='ban') {
      const mins = parseInt(prompt('Minutes (0 = perm)'))||0;
      const reason = prompt('Reason');
      await Nui('players:ban', { target, minutes: mins, reason });
    }
    if(act==='kick') {
      const reason = prompt('Reason');
      await Nui('players:kick', { target, reason });
    }
  });

  // Vehicle
  $('#btn-veh-spawn').on('click', async ()=>{ await Nui('vehicle:spawn', { model: $('#veh-model').val() }); });
  $('#btn-veh-fix').on('click', async ()=>{ await Nui('vehicle:fix', {}); });
  $('#btn-veh-clean').on('click', async ()=>{ await Nui('vehicle:clean', {}); });
  $('#btn-veh-delnear').on('click', async ()=>{ await Nui('vehicle:deleteNear', { radius: 8.0 }); });

  // Main
  $('#btn-announce').on('click', async ()=>{ await Nui('world:announce', { msg: $('#announce-msg').val() }); });
  $('#btn-weather').on('click', async ()=>{ await Nui('world:weather', { weather: $('#weather').val(), duration: parseInt($('#weather-duration').val())||0 }); });
  $('#btn-time').on('click', async ()=>{ await Nui('world:time', { hour: parseInt($('#time-hour').val())||12, minute: parseInt($('#time-minute').val())||0, scale: parseFloat($('#time-scale').val())||1.0 }); });

  // Settings
  $('#btn-save-hotkey').on('click', async ()=>{ await Nui('settings:hotkey', { key: $('#key-open').val() }); });

  // Personal toggles
  $('[data-per]').on('click', async function(){
    const k = $(this).data('per');
    if(k==='noclip') await Nui('personal:noclip');
    if(k==='invisible') await Nui('personal:invisible');
    if(k==='superjump') await Nui('personal:superjump');
    if(k==='fastrun') await Nui('personal:fastrun');
    if(k==='ghost') await Nui('personal:ghost');
```
